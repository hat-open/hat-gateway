128    async def _process_data(self, version, subkey, data):
129        try:
130            events = collections.deque()
131
132            for key in [(version, subkey),
133                        (version, None)]:
134                for remote_device in self._remote_devices.get(key, []):
135                    for i in data:
136                        if i.name not in remote_device.oids:
137                            continue
138
139                        event = hat.event.common.RegisterEvent(
140                            type=(*self._event_type_prefix, 'gateway', 'data',
141                                  remote_device.name, _oid_to_str(i.name)),
142                            source_timestamp=None,
143                            payload=hat.event.common.EventPayloadJson(
144                                _event_payload_from_data(
145                                    i, remote_device.string_hex_oids)))
146                        events.append(event)
147
148            if not events:
149                return
150
151            await self._eventer_client.register(events)
152
153        except ConnectionError:
154            pass
155
156        except Exception as e:
157            self._log.error("error processing data: %s", e, exc_info=e)

174def _event_payload_from_data(data, string_hex_oids):
175    if isinstance(data, snmp.EmptyData):
176        return {'type': 'ERROR',
177                'value': 'EMPTY'}
178
179    if isinstance(data, snmp.UnspecifiedData):
180        return {'type': 'ERROR',
181                'value': 'UNSPECIFIED'}
182
183    if isinstance(data, snmp.NoSuchObjectData):
184        return {'type': 'ERROR',
185                'value': 'NO_SUCH_OBJECT'}
186
187    if isinstance(data, snmp.NoSuchInstanceData):
188        return {'type': 'ERROR',
189                'value': 'NO_SUCH_INSTANCE'}
190
191    if isinstance(data, snmp.EndOfMibViewData):
192        return {'type': 'ERROR',
193                'value': 'END_OF_MIB_VIEW'}
194
195    if isinstance(data, snmp.IntegerData):
196        return {'type': 'INTEGER',
197                'value': data.value}
198
199    if isinstance(data, snmp.UnsignedData):
200        return {'type': 'UNSIGNED',
201                'value': data.value}
202
203    if isinstance(data, snmp.CounterData):
204        return {'type': 'COUNTER',
205                'value': data.value}
206
207    if isinstance(data, snmp.BigCounterData):
208        return {'type': 'BIG_COUNTER',
209                'value': data.value}
210
211    if isinstance(data, snmp.TimeTicksData):
212        return {'type': 'TIME_TICKS',
213                'value': data.value}
214
215    if isinstance(data, snmp.StringData):
216        if data.name in string_hex_oids:
217            return {'type': 'STRING_HEX',
218                    'value': data.value.hex()}
219
220        return {'type': 'STRING',
221                'value': str(data.value, encoding='utf-8', errors='replace')}
222
223    if isinstance(data, snmp.ObjectIdData):
224        return {'type': 'OBJECT_ID',
225                'value': _oid_to_str(data.value)}
226
227    if isinstance(data, snmp.IpAddressData):
228        return {'type': 'IP_ADDRESS',
229                'value': '.'.join(str(i) for i in data.value)}
230
231    if isinstance(data, snmp.ArbitraryData):
232        return {'type': 'ARBITRARY',
233                'value': data.value.hex()}
234
235    raise Exception('invalid response data')
