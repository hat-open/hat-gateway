175def _msg_from_event(event_type_prefix, message_encoding, event):
176    suffix = event.type[len(event_type_prefix):]
177    if suffix != ('system', 'message'):
178        raise Exception('unsupported event type')
179
180    if not isinstance(event.payload.data['address'], str):
181        raise Exception('invalid address type')
182
183    if not isinstance(event.payload.data['message'], str):
184        raise Exception('invalid message type')
185
186    return _Msg(address=event.payload.data['address'],
187                message=event.payload.data['message'].encode(message_encoding))
