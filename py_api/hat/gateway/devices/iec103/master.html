218    async def _process_event(self, event):
219        prefix_len = len(self._event_type_prefix)
220        if event.type[prefix_len + 1] != 'remote_device':
221            raise Exception('unexpected event type')
222
223        address = self._address_from_event(event)
224        etype_suffix = event.type[prefix_len + 3:]
225
226        if etype_suffix[0] == 'enable':
227            self._process_enable(event)
228
229        elif etype_suffix[0] == 'command':
230            asdu = int(etype_suffix[1])
231            io = iec103.IoAddress(
232                function_type=int(etype_suffix[2]),
233                information_number=int(etype_suffix[3]))
234            self._process_command(event, address, asdu, io)
235
236        elif etype_suffix[0] == 'interrogation':
237            asdu = int(etype_suffix[1])
238            self._process_interrogation(event, address, asdu)
239
240        else:
241            raise Exception('unexpected event type')

243    def _process_enable(self, event):
244        address = self._address_from_event(event)
245        enable = event.payload.data
246        if address not in self._remote_enabled:
247            raise Exception('invalid remote device address')
248
249        if not isinstance(enable, bool):
250            raise Exception('invalid enable event payload')
251
252        self._remote_enabled[address] = enable
253
254        if not self._master:
255            return

257        if enable:
258            self._enable_remote(address)

260        else:

263    def _enable_remote(self, address):
264        remote_group = self._remote_groups.get(address)
265        if remote_group and remote_group.is_open:
266            return

273        if address in self._remote_groups:

364def _events_from_data(data, address, event_type_prefix):
365    cause = (data.cause.name if isinstance(data.cause, enum.Enum)
366             else data.cause)
367
368    if isinstance(data.value, (iec103.DoubleWithTimeValue,
369                               iec103.DoubleWithRelativeTimeValue)):
370        data_type = 'double'
371        payload = {'cause': cause,
372                   'value': data.value.value.name}
373        source_ts = _time_iec103_to_source_ts(data.value.time)
374        event_type = _data_event_type(
375            data, address, data_type, event_type_prefix)
376        yield _create_event(event_type, payload, source_ts)
377
378    elif isinstance(data.value, iec103.MeasurandValues):
379        for meas_type, meas_value in data.value.values.items():
380            payload = {'cause': cause,
381                       'value': meas_value._asdict()}
382            data_type = meas_type.name.lower()
383            event_type = _data_event_type(
384                data, address, data_type, event_type_prefix)
385            yield _create_event(event_type, payload)
386
387    else:
388        raise Exception('unsupported data value')

400def _time_iec103_to_source_ts(time_four):
401    t_now = datetime.datetime.now(datetime.timezone.utc)
402    candidates_now = [t_now - datetime.timedelta(hours=12),
403                      t_now,
404                      t_now + datetime.timedelta(hours=12)]
405    candidates_103 = [_upgrade_time_four_to_seven(
406                        time_four, iec103.time_from_datetime(t))
407                      for t in candidates_now]
408    candidates_dt = [iec103.time_to_datetime(t)
409                     for t in candidates_103 if t]
410    if not candidates_dt:
411        return
412
413    res = min(candidates_dt, key=lambda i: abs(t_now - i))
414    return hat.event.common.timestamp_from_datetime(res)

417def _upgrade_time_four_to_seven(time_four, time_seven):
418    if time_four.summer_time != time_seven.summer_time:
419        return
420
421    return time_four._replace(
422        day_of_week=time_seven.day_of_week,
423        day_of_month=time_seven.day_of_month,
424        months=time_seven.months,
425        years=time_seven.years,
426        size=iec103.TimeSize.SEVEN)
